 22 /**
 23  * Extends constructor function (class) from parent constructor using prototype inherinatce
 24  * @public
 25  * @param {function} child Child class
 26  * @param {function} parent Parent class
 27  */
 28 kff.extend = function(child, parent)
 29 {
 30 	var F = function(){};
 31 	child._super = F.prototype = parent.prototype;
 32 	child.prototype = new F();
 33 	child.prototype.constructor = child;
 34 };

 36 /**
 37  * Mixins (using a shallow copy) properties from one object to another
 38  * Function accepts multiple arguments with multiple extending objects
 39  * When passing true as the last argument, deep copy will be used
 40  *
 41  * @param {Object} obj Object to be extended
 42  * @param {Object} properties Extending object(s)
 43  * @returns {Object} Extended object (=== obj)
 44  */
 45 kff.mixins = function(obj, properties)
 46 {
 47 	var i = 1, l = arguments.length, key, props, prop, objProp, deep = false;
 48 	if(l > 2 && arguments[l-1] === true)
 49 	{
 50 		deep = true;
 51 		l--;
 52 	}
 53 	while(i < l)
 54 	{
 55 		props = arguments[i];
 56 		for(key in props)
 57 		{
 58 			if(props.hasOwnProperty(key))
 59 			{
 60 				prop = props[key];
 61 				if(deep && kff.isPlainObject(prop))
 62 				{
 63 					objProp = obj[key];
 64 					if(typeof objProp !== 'object' || objProp === null) objProp = {};
 65 					kff.mixins(objProp, prop, deep);
 66 				}
 67 				else obj[key] = prop;
 68 			}
 69 		}
 70 		i++;
 71 	}
 72 	return obj;
 73 };

 75 /**
 76  * Factory for creating a class
 77  * @param {Object} meta Object with metadata describing inheritance and static properties of the class
 78  * @param {Object} properties Properties of a class prototype (or class members)
 79  * @returns {function} A constructor function (class)
 80  */
 81 kff.createClass = function(meta, properties)
 82 {
 83 	var constructor;
 84 	if(arguments.length === 0) meta = properties = {};
 85 	else if(arguments.length === 1)
 86 	{
 87 		properties = meta;
 88 		meta = {};
 89 	}
 90 
 91 	// Create a new constructor if not defined in properties
 92 	if(properties.hasOwnProperty('constructor'))
 93 	{
 94 		constructor = properties.constructor;
 95 	}
 96 	else
 97 	{
 98 		if(meta.extend) constructor = function(){ meta.extend.apply(this, arguments); };
 99 		else constructor = function(){};
100 	}
101 
102 	// Extend from parent class
103 	if(meta.extend) kff.extend(constructor, meta.extend);
104 
105 	// Concatenate properties from properties objects and mixin objects
106 	if(!('mixins' in meta))
107 	{
108 		meta.mixins = [];
109 	}
110 	else if(!(meta.mixins instanceof Array)) meta.mixins = [meta.mixins];
111 
112 	meta.mixins.push(kff.classMixin);
113 
114 	for(var i = 0, l = meta.mixins.length; i < l; i++) kff.mixins(properties, meta.mixins[i]);
115 
116 	// Static properties of constructor
117 	if(meta.staticProperties)
118 	{
119 		kff.mixins(constructor, meta.staticProperties);
120 	}
121 
122 	// Add properties to prototype
123 	kff.mixins(constructor.prototype, properties);
124 
125 	// Set proper constructor
126 	constructor.prototype.constructor = constructor;
127 
128 	return constructor;
129 };

131 /**
132  * Binds function to an object. Adds _boundFns object width references to bound methods for caching purposes.
133  * @param {Object} obj Object to which bind a function
134  * @param {string} fnName Method name to bind
135  */
136 kff.bindFn = function(obj, fnName)
137 {
138 	if(typeof obj[fnName] !== 'function') throw new TypeError("Expected function: " + fnName + ' (kff.bindFn)');
139 	if(!('_boundFns' in obj)) obj._boundFns = {};
140 	if(fnName in obj._boundFns) return obj._boundFns[fnName];
141 	else obj._boundFns[fnName] = function(){ return obj[fnName].apply(obj, arguments); };
142 	return obj._boundFns[fnName];
143 };

146 	f: function(fnName)
147 	{
148 		var obj = this;
149 		if(typeof fnName === 'string') return kff.bindFn(obj, fnName);
150 		if(typeof fnName === 'function') return function(){ return fnName.apply(obj, arguments); };
151 		throw new TypeError("Expected function: " + fnName + ' (kff.f)');
152 	}

155 /**
156  * Evaluates object path recursively and returns last property in chain
157  *
158  * Example:
159  * window.something = { foo: { bar: 42 } };
160  * kff.evalObjectPath('something.foo.bar', window) === 42 // true
161  *
162  * @param {string} path object path (like 'something.foo.bar')
163  * @param {Object} obj Object to start with (like window)
164  * @returns {mixed} Property at the end of object chain or null if not found
165  */
166 kff.evalObjectPath = function(path, obj)
167 {
168 	obj = obj || scope;
169 	var parts = path.split('.');
170 	while(parts.length)
171 	{
172 		if(!(parts[0] in obj)) return null;
173 		obj = obj[parts.shift()];
174 	}
175 	return obj;
176 };

185 	return obj !== null && typeof obj === 'object' && obj.constructor === Object;

190 kff.Events = kff.createClass(
191 /** @lends kff.Events.prototype */
192 {
193 	/**
194 		@constructs
195 	*/
196 	constructor: function()
197 	{
198 		this.subscribers = {};
199 		this.oneSubscribers = {};
200 	},
201 
202 	/**
203 		Binds event handler.
204 
205 		@param {string|Array} eventType Event name(s)
206 		@param {function} fn Event handler
207 	*/
208 	on: function(eventType, fn)
209 	{
210 		this.off(eventType, fn);
211 		if(eventType instanceof Array)
212 		{
213 			for(var i = 0, l = eventType.length; i < l; i++)
214 			{
215 				if(eventType[i])
216 				{
217 					if(!this.subscribers[eventType[i]]) this.subscribers[eventType[i]] = [];
218 					this.subscribers[eventType[i]].push(fn);
219 				}
220 			}
221 		}
222 		else
223 		{
224 			if(!this.subscribers[eventType]) this.subscribers[eventType] = new kff.List();
225 			this.subscribers[eventType].append(fn);
226 		}
227 	},
228 
229 	/**
230 		Binds event handler that will be executed only once.
231 
232 		@param {string|Array} eventType Event name(s)
233 		@param {function} fn Event handler
234 	*/
235 	one: function(eventType, fn)
236 	{
237 		if(!(eventType in this.oneSubscribers)) this.oneSubscribers[eventType] = [];
238 		this.oneSubscribers[eventType].push(fn);
239 		this.on(eventType, fn);
240 	},
241 
242 	/**
243 		Unbinds event handler.
244 
245 		@param {string|Array} eventType Event name(s)
246 		@param {function} fn Event handler
247 	*/
248 	off: function(eventType, fn)
249 	{
250 		var i, l;
251 		if(eventType instanceof Array)
252 		{
253 			for(i = 0, l = eventType.length; i < l; i++)
254 			{
255 				if(eventType[i]) this.off(eventType[i], fn);
256 			}
257 		}
258 		else
259 		{
260 			if(this.subscribers[eventType] instanceof kff.List) this.subscribers[eventType].removeVal(fn);
261 		}
262 	},
263 
264 	/**
265 		Triggers an event.
266 
267 		@param {string|Array} eventType Event name(s)
268 		@param {mixed} eventData Arbitrary data that will be passed to the event handlers as an argument
269 	*/
270 	trigger: function(eventType, eventData)
271 	{
272 		var i, l;
273 		if(eventType instanceof Array)
274 		{
275 			for(i = 0, l = eventType.length; i < l; i++)
276 			{
277 				if(eventType[i]) this.trigger(eventType[i], eventData);
278 			}
279 		}
280 		else
281 		{
282 			if(this.subscribers[eventType] instanceof kff.List)
283 			{
284 				this.subscribers[eventType].each(function(val)
285 				{
286 					val.call(null, eventData);
287 				});
288 
289 				// Remove "one" subscribers:
290 				if(eventType in this.oneSubscribers)
291 				{
292 					for(i = 0, l = this.oneSubscribers[eventType].length; i < l; i++)
293 					{
294 						this.off(eventType, this.oneSubscribers[eventType][i]);
295 					}
296 					this.oneSubscribers[eventType] = [];
297 				}
298 			}
299 		}
300 	}
301 });

317 /** @lends kff.List.prototype */
318 {
319 	/**
320 		Class representing a list data structure
321 		@constructs
322 	 */
323 	constructor: function()
324 	{
325 		this.array = [];
326 		this.count = this.array.length;
327 	},
328 
329 	/**
330 		Iterates over each item in the list
331 		@param {function} fn A callback function to be called on each item. Takes two arguments - the iterated item and its index
332 	 */
333 	each: function(fn)
334 	{
335 		var a = this.array, l = a.length, i = 0;
336 		for(; i < l; i++)
337 		{
338 			if(fn.call(null, a[i], i) === false) break;
339 		}
340 	},
341 
342 	/**
343 		Appends an item at the end of the list
344 		@param {mixed} val Item to be appended
345 	 */
346 	append: function(val)
347 	{
348 		this.array.push(val);
349 		this.count++;
350 	},
351 
352 	/**
353 		Inserts an item at specified index
354 		@param {mixed} val Item to be inserted
355 	 */
356 	insert: function(val, index)
357 	{
358 		this.array.splice(index, 0, val);
359 		this.count++;
360 	},
361 
362 	/**
363 		Removes item from the list
364 		@param {mixed} val Reference to the item to be removed
365 		@returns {Boolean} True if item was removed or false if not found
366 	 */
367 	remove: function(val)
368 	{
369 		var i = this.indexOf(val);
370 		if(i === -1) return false;
371 
372 		this.array.splice(i, 1);
373 		this.count--;
374 
375 		return true;
376 	},
377 
378 	/**
379 		Removes all items from the list
380 	 */
381 	empty: function()
382 	{
383 		this.array = [];
384 		this.count = 0;
385 	},
386 
387 	/**
388 		Returns an index of given item
389 
390 		@param {mixed} val Value to be found
391 		@returns {number} index of the item or -1 if not found
392 	 */
393 	indexOf: function(val)
394 	{
395 		var i = 0, a = this.array, l = a.length;
396 		if(a.indexOf) return a.indexOf(val);
397 		for(; i < l; i++) if(a[i] === val) return i;
398 		return -1;
399 	},
400 
401 	/**
402 		Sets an item at given position
403 
404 		@param {number} index Index of item
405 		@param {mixed} item Item to set
406 	 */
407 	set: function(index, item)
408 	{
409 		if(this.array[index] !== undefined)	this.array[index] = item;
410 		else throw new RangeError('Bad index in kff.List.set');
411 	},
412 
413 	/**
414 		Returns an item at given position
415 
416 		@param {number} index Index of item
417 		@returns {mixed} Item at given position (or undefined if not found)
418 	 */
419 	get: function(index)
420 	{
421 		return this.array[index];
422 	},
423 
424 	/**
425 		Sorts list using a compare function. The compare function follows the same specification
426 		as the standard Array.sort function
427 
428 		@param {function} compareFunction Compare function
429 	 */
430 	sort: function(compareFunction)
431 	{
432 		this.array.sort(compareFunction);
433 	},
434 
435 	/**
436 		Randomizes items in the list.
437 	 */
438 	shuffle: function()
439 	{
440 		var arr = this.array,
441 			len = arr.length,
442 			i = len,
443 			p, t;
444 
445 		while(i--)
446 		{
447 			p = parseInt(Math.random()*len, 10);
448 			t = arr[i];
449 			arr[i] = arr[p];
450 			arr[p] = t;
451 		}
452 	}
453 
454 });

699

701 		@param {string} attr Attribute name
702 		@returns {boolean} True if found, false otherwise

711

713 		@returns {mixed} Attribute value
714 	 */
715 	get: function(attr)
716 	{
717 		return this.attrs[attr];

719 
720 	/**
721 		Returns the value of given attribute using deep lookup (object.attribute.some.value)
722 
723 		@param {string} attrPath Attribute path
724 		@returns {mixed} Attribute value
725 	 	@example
726 	 	obj.mget('one.two.three');
727 	 	// equals to:
728 	 	obj.get('one').get('two').get('three');
729 	 */
730 	mget: function(attrPath)
731 	{
732 		var attr;
733 		if(typeof attrPath === 'string') attrPath = attrPath.split('.');
734 		attr = this.get(attrPath.shift());
735 		if(attrPath.length > 0)
736 		{
737 			if(attr instanceof kff.Model || attr instanceof kff.Collection) return attr.mget(attrPath);
738 			else return kff.evalObjectPath(attrPath, attr);
739 		}
740 		else return attr;
741 	},

743 	/**
744 		Sets the value(s) of given attribute(s). Triggers change event.
745 
746 		@param {string} attr Attribute name
747 		@param {mixed} value Attribute value
748 		@param {Boolean} silent If true, do not trigger event
749 	 */
750 	set: function(attr, value, silent)
751 	{
752 		var changed = {};
753 
754 		if(typeof attr === 'string')
755 		{
756 			if(this.get(attr) === value) return;
757 			changed[attr] = value;
758 			this.attrs[attr] = value;
759 		}
760 		else if(attr !== null && attr instanceof Object)
761 		{
